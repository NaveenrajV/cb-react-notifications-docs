   return  
    <Notifications 
        data={data} 
        height='500px'
        width='300px'
        classNamePrefix='okrjoy'

        headerBackgroundColor = 'red'
        header={
          {
            title: 'Notifications',
            option: { text: 'View All', onClick: () => {} }
          }
        }

    />
}
